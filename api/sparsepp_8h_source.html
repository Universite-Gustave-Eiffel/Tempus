1247 // the stream argument twice: once as a template type, and once with
1248 // normal type information. (We only use the second version.) We do
1249 // this because of how C++ picks what function overload to use. If we
1250 // implemented this the naive way:
1251 // bool read_data(istream* is, const void* data, size_t length);
1252 // template<typename T> read_data(T* fp, const void* data, size_t length);
1253 // C++ would prefer the second version for every stream type except
1254 // istream. However, we want C++ to prefer the first version for
1255 // streams that are *subclasses* of istream, such as istringstream.
1256 // This is not possible given the way template types are resolved. So
1257 // we split the stream argument in two, one of which is templated and
1258 // one of which is not. The specialized functions (like the istream
1259 // version above) ignore the template arg and use the second, 'type'
1260 // arg, getting subclass matching as normal. The 'catch-all'
1261 // functions (the second version above) use the template arg to deduce
1262 // the type, and use a second, void* arg to achieve the desired

1350  // length is the number of bytes to read/write (which may differ
1351  // from sizeof(IntType), allowing us to save on a 32-bit system
1352  // and load on a 64-bit system). Excess bytes are taken to be 0.

1354  // --------------------------------------------------------------------
1355  template <typename INPUT, typename IntType>
1356  bool read_bigendian_number(INPUT* fp, IntType* value, size_t length)
1357  {
1358  *value = 0;
1359  unsigned char byte;
1360  // We require IntType to be unsigned or else the shifting gets all screwy.
1361  SPP_COMPILE_ASSERT(static_cast<IntType>(-1) > static_cast<IntType>(0), "serializing_int_requires_an_unsigned_type");

1363  {
1364  if (!read_data(fp, &byte, sizeof(byte)))
1365  return false;
1366  *value |= static_cast<IntType>(byte) << ((length - 1 - i) * 8);
1367  }
1368  return true;
1369  }

1549 // S P A R S E T A B L E
1550 // ----------------------------------------------------------------------
1551 //
1552 // A sparsetable is a random container that implements a sparse array,
1553 // that is, an array that uses very little memory to store unassigned
1554 // indices (in this case, between 1-2 bits per unassigned index). For
1555 // instance, if you allocate an array of size 5 and assign a[2] = <big
1556 // struct>, then a[2] will take up a lot of memory but a[0], a[1],
1557 // a[3], and a[4] will not. Array elements that have a value are
1558 // called "assigned". Array elements that have no value yet, or have
1559 // had their value cleared using erase() or clear(), are called
1560 // "unassigned".
1561 //
1562 // Unassigned values seem to have the default value of T (see below).
1563 // Nevertheless, there is a difference between an unassigned index and
1564 // one explicitly assigned the value of T(). The latter is considered
1565 // assigned.
1566 //
1567 // Access to an array element is constant time, as is insertion and
1568 // deletion. Insertion and deletion may be fairly slow, however:
1569 // because of this container's memory economy, each insert and delete
1570 // causes a memory reallocation.
1571 //
1572 // NOTE: You should not test(), get(), or set() any index that is
1573 // greater than sparsetable.size(). If you need to do that, call
1574 // resize() first.
1575 //
1576 // --- Template parameters
1577 // PARAMETER DESCRIPTION DEFAULT
1578 // T The value of the array: the type of --
1579 // object that is stored in the array.
1580 //
1581 // Alloc: Allocator to use to allocate memory. libc_allocator_with_realloc
1582 //
1583 // --- Model of
1584 // Random Access Container
1585 //
1586 // --- Type requirements
1587 // T must be Copy Constructible. It need not be Assignable.
1588 //
1589 // --- Public base classes
1590 // None.
1591 //
1592 // --- Members
1593 //
1594 // [*] All iterators are const in a sparsetable (though nonempty_iterators
1595 // may not be). Use get() and set() to assign values, not iterators.
1596 //
1597 // [+] iterators are random-access iterators. nonempty_iterators are
1598 // bidirectional iterators.
1599 
1600 // [*] If you shrink a sparsetable using resize(), assigned elements
1601 // past the end of the table are removed using erase(). If you grow
1602 // a sparsetable, new unassigned indices are created.
1603 //
1604 // [+] Note that operator[] returns a const reference. You must use
1605 // set() to change the value of a table element.
1606 //
1607 // [!] Unassignment also calls the destructor.
1608 //
1609 // Iterators are invalidated whenever an item is inserted or
1610 // deleted (ie set() or erase() is used) or when the size of
1611 // the table changes (ie resize() or clear() is used).

2271 // SPARSE-TABLE
2272 // ------------
2273 // The idea is that a table with (logically) t buckets is divided
2274 // into t/M *groups* of M buckets each. (M is a constant, typically
2275 // 32) Each group is stored sparsely.
2276 // Thus, inserting into the table causes some array to grow, which is
2277 // slow but still constant time. Lookup involves doing a
2278 // logical-position-to-sparse-position lookup, which is also slow but
2279 // constant time. The larger M is, the slower these operations are
2280 // but the less overhead (slightly).
2281 //
2282 // To store the sparse array, we store a bitmap B, where B[i] = 1 iff
2283 // bucket i is non-empty. Then to look up bucket i we really look up
2284 // array[# of 1s before i in B]. This is constant time for fixed M.
2285 //
2286 // Terminology: the position of an item in the overall table (from
2287 // 1 .. t) is called its "location." The logical position in a group
2288 // (from 1 .. M) is called its "position." The actual location in
2289 // the array (from 1 .. # of non-empty buckets in the group) is
2290 // called its "offset."
2291 // ---------------------------------------------------------------------------

2368  {
2369 #if !defined(SPP_ALLOC_SZ) || (SPP_ALLOC_SZ == 0)
2370  // aggressive allocation first, then decreasing as sparsegroups fill up
2371  // --------------------------------------------------------------------
2372  static uint8_t s_alloc_batch_sz[SPP_GROUP_SIZE] = { 0 };
2373  if (!s_alloc_batch_sz[0])
2374  {
2375  // 32 bit bitmap
2376  // ........ .... .... .. .. .. .. . . . . . . . .
2377  // 8 12 16 18 20 22 24 25 26 ... 32
2378  // ------------------------------------------------------
2379  uint8_t group_sz = SPP_GROUP_SIZE / 4;
2380  uint8_t group_start_alloc = SPP_GROUP_SIZE / 8; //4;
2381  uint8_t alloc_sz = group_start_alloc;

2383  {
2384  for (int j=0; j<group_sz; ++j)
2385  {
2386  if (j && j % group_start_alloc == 0)
2387  alloc_sz += group_start_alloc;
